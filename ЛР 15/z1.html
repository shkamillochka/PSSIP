2. Доступ к свойствам объекта

const book = { title: "JavaScript: The Good Parts", author: "Douglas Crockford", year: 2008, "is-available": true // Свойство с дефисом }; // Точечная нотация console.log(book.title); // Скобочная нотация console.log(book["author"]); console.log(book["is-available"]); // Обязательно для спецсимволов // Динамический доступ const prop = "year"; console.log(book[prop]);

3. Удаление свойств (delete)

const product = { name: "Ноутбук", price: 50000, brand: "Lenovo", warranty: 2 }; console.log("До удаления:", Object.keys(product)); delete product.warranty; console.log("После удаления:", Object.keys(product));

4. Проверка существования свойств

const user = { username: "john_doe", email: "john@example.com", age: 30 }; // Оператор in console.log("username" in user); // true console.log("phone" in user); // false // Метод hasOwnProperty() console.log(user.hasOwnProperty("email")); // true console.log(user.hasOwnProperty("address")); // false // Проверка через undefined console.log(user.age !== undefined); // true console.log(user.city !== undefined); // false

5. Перебор свойств объекта

const car = { brand: "BMW", model: "X5", year: 2023, color: "black", price: 7500000 }; // for...in for (let key in car) { console.log(`${key}: ${car[key]}`); } // Object.keys() const keys = Object.keys(car); console.log("Ключи:", keys); // Object.values() const values = Object.values(car); console.log("Значения:", values); // Object.entries() Object.entries(car).forEach(([key, value]) => { console.log(`Свойство "${key}" имеет значение "${value}"`); });

6. Создание объектов через конструктор

// Функция-конструктор function Person(name, age, city) { this.name = name; this.age = age; this.city = city; this.getInfo = function() { return `${this.name}, ${this.age} лет, г. ${this.city}`; }; } // Создание экземпляров const person1 = new Person("Мария", 25, "Санкт-Петербург"); const person2 = new Person("Иван", 30, "Москва"); console.log(person1.getInfo()); console.log(person2.getInfo());

7. Классы в JavaScript (ES6)

class Animal { constructor(name, type, sound) { this.name = name; this.type = type; this.sound = sound; } // Метод экземпляра makeSound() { return `${this.name} говорит: ${this.sound}!`; } // Статический метод static isAnimal(obj) { return obj instanceof Animal; } } // Наследование class Dog extends Animal { constructor(name, breed) { super(name, "собака", "Гав"); this.breed = breed; } bark() { return `${this.name} лает: ${this.sound} ${this.sound}!`; } } const myDog = new Dog("Бобик", "овчарка"); console.log(myDog.makeSound()); console.log(myDog.bark()); console.log(Animal.isAnimal(myDog));
